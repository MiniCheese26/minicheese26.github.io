using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;
using System.Text.RegularExpressions;

namespace BatchRenamer
{
    class Operations
    {
        public static void BatchRename()
        {
            Console.WriteLine("Batch Rename");
            Console.WriteLine("----------------------");
            Console.WriteLine("Enter your directory");
            string dir = Console.ReadLine();

            while (!Directory.Exists(dir))
            {
                Console.WriteLine("");
                Console.WriteLine("Directory does not exist. Try again");
                dir = Console.ReadLine();
            }

            Console.WriteLine("What would you like to batch rename these files?");
            string rename = Console.ReadLine();

            rename = dir + @"\" + rename;
            int i = 0;
            int renameLength = rename.Length;
            string renameRaw = rename;

            string[] files = Directory.GetFiles(dir);

            foreach (string file in files)
            {
                string fileName = Path.GetFileName(file);

                string[] extensionArray = fileName.Split('.');
                string extension = extensionArray[1];

                if (i != 0)
                {
                    rename = renameRaw + i;
                }

                string renameFinal = rename + "." + extension;

                File.Move(file, renameFinal);
                i++;
            }

            Console.WriteLine("");
            Console.WriteLine("Done");
            Console.ReadKey();
            Console.Clear();
            Program.Main();
        }
    }
}
